using System;
using System.Diagnostics;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;

namespace CrashTest.Unified
{
    public class CrashSelectorForm : Form
    {
        public CrashSelectorForm()
        {
            InitializeComponents();
        }

        private void InitializeComponents()
        {
            Text = "CrashTest Unified - бЁдёБРРЃРраЭ";
            Size = new Size(700, 600);
            StartPosition = FormStartPosition.CenterScreen;
            
            var mainLayout = new TableLayoutPanel
            {
                Dock = DockStyle.Fill,
                RowCount = 3,
                Padding = new Padding(10)
            };
            mainLayout.RowStyles.Add(new RowStyle(SizeType.Absolute, 60));
            mainLayout.RowStyles.Add(new RowStyle(SizeType.Percent, 100));
            mainLayout.RowStyles.Add(new RowStyle(SizeType.Absolute, 40));

            // БъЬт
            var title = new Label
            {
                Text = "?? WinForm ЩСЭЫВтЪдЬзМў",
                Font = new Font("Microsoft YaHei", 16, FontStyle.Bold),
                Dock = DockStyle.Fill,
                TextAlign = ContentAlignment.MiddleCenter
            };
            mainLayout.Controls.Add(title, 0, 0);

            // АДХЅЭјИё
            var buttonPanel = new FlowLayoutPanel
            {
                Dock = DockStyle.Fill,
                AutoScroll = true,
                FlowDirection = FlowDirection.LeftToRight,
                WrapContents = true
            };

            var crashes = CrashTrigger.GetAvailableCrashes();
            foreach (var (key, (name, desc, exitCode)) in crashes)
            {
                var btn = new Button
                {
                    Text = $"{key.ToUpper()}\n{name}\n{desc}",
                    Size = new Size(200, 100),
                    Margin = new Padding(5),
                    Font = new Font("Consolas", 9),
                    BackColor = GetColorForCrashType(key),
                    ForeColor = Color.White,
                    FlatStyle = FlatStyle.Flat
                };
                
                btn.Click += (s, e) => 
                {
                    var result = MessageBox.Show(
                        $"ШЗЖЈвЊДЅЗЂ [{key.ToUpper()}] {name}?\n\n" +
                        $"УшЪі: {desc}\n" +
                        $"дЄЦкЭЫГіТы: 0x{exitCode:X8}\n\n" +
                        "етНЋЕМжТГЬађСЂМДБРРЃЃЁ",
                        "ШЗШЯБРРЃВтЪд",
                        MessageBoxButtons.YesNo,
                        MessageBoxIcon.Warning);
                    
                    if (result == DialogResult.Yes)
                    {
                        CrashTrigger.Execute(key);
                    }
                };
                
                // ЬэМгаќЭЃЬсЪО
                var toolTip = new ToolTip();
                toolTip.SetToolTip(btn, $"УќСюаа: CrashTest.Unified.exe {key}\n\n{desc}");
                
                buttonPanel.Controls.Add(btn);
            }
            
            mainLayout.Controls.Add(buttonPanel, 0, 1);

            // ЕзВПаХЯЂ

            {
                Text = "ЬсЪО: вВПЩЭЈЙ§УќСюааВЮЪ§ЦєЖЏЃЌШч: CrashTest.Unified.exe av",
                Dock = DockStyle.Fill,
                TextAlign = ContentAlignment.MiddleCenter,
                ForeColor = Color.Gray
            };
            mainLayout.Controls.Add(info, 0, 2);

            Controls.Add(mainLayout);
        }

        private Color GetColorForCrashType(string key) => key switch
        {
            "av" => Color.DarkRed,
            "so" => Color.DarkOrange,
            "hc" => Color.Purple,
            "dz" => Color.DarkBlue,
            "nr" => Color.DarkCyan,
            "oom" => Color.DarkGreen,
            "dl" => Color.DarkGoldenrod,
            "md" => Color.DarkMagenta,
            _ => Color.DarkGray
        };
    }
}
