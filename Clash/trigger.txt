// ==================== 17. ЯпГЬОКељ/ИпВЂЗЂЫјељгУ ====================
private static void TriggerThreadContention()
{
    var sharedResource = new object();
    var threads = new List<Thread>();
    var cpuCount = Environment.ProcessorCount;
    
    Console.WriteLine($"ЦєЖЏ {cpuCount * 4} ИіЯпГЬНјааЫјОКељ...");
    
    // ДДНЈДѓСПЯпГЬОКељЭЌвЛАбЫј
    for (int i = 0; i < cpuCount * 4; i++)
    {
        var thread = new Thread(() =>
        {
            while (true)
            {
                // МЋЖШЦЕЗБЕФЫјЧаЛЛ
                lock (sharedResource)
                {
                    // ГжгаЫјЪБзівЛЕуЙЄзї
                    for (int j = 0; j < 1000; j++)
                    {
                        Math.Sqrt(j);
                    }
                }
                // СЂМДдйДЮГЂЪдЛёШЁЫјЃЌдьГЩМЄСвОКељ
            }
        });
        thread.Priority = ThreadPriority.Highest;
        threads.Add(thread);
        thread.Start();
    }

    // жїЯпГЬвВВЮгыОКељЃЌЕМжТUIЭъШЋПЈЫР
    while (true)
    {
        lock (sharedResource)
        {
            Thread.Sleep(1);
        }
    }
}

// COMЗУЮЪГхЭЛ - ЭЈЙ§ВЛАВШЋЕФCOMЛЅВйзї
private static unsafe void COMCrash_AccessViolation()
{
    // ГЂЪдЪжЖЏЕїгУCOMащБэЃЌДЋШыЮоаЇжИеы
    // ФЃФтCOMзщМўФкВПБРРЃ

    var shell = new Shell32.Shell(); // Windows Shell COMЖдЯѓ
    var punk = Marshal.GetIUnknownForObject(shell);
    
    // ЛёШЁащБэжИеы
    var vtbl = *(IntPtr**)punk;
    
    // ЕїгУащБэжаЕФЗНЗЈЃЌЕЋДЋШыЭъШЋЮоаЇЕФВЮЪ§
    // етЛсЕМжТCOMзщМўФкВПЗУЮЪГхЭЛ
    var releaseDelegate = Marshal.GetDelegateForFunctionPointer<ReleaseDelegate>(vtbl[2]);
    
    // ЯШе§ГЃЪЭЗХвЛДЮ
    releaseDelegate(punk);
    
    // дйДЮЪЭЗХЭЌвЛИіжИеыЃЈЫЋжиЪЭЗХCOMЖдЯѓЃЉ
    releaseDelegate(punk); // етРяЛсЕМжТCOMЖдЯѓЖбЫ№ЛЕ
    
    // ЛђепГЂЪдЭЈЙ§вбЪЭЗХЕФжИеыЕїгУЗНЗЈ
    var queryInterface = Marshal.GetDelegateForFunctionPointer<QueryInterfaceDelegate>(vtbl[0]);
    Guid iid = Guid.Parse("00000000-0000-0000-C000-000000000046");
    IntPtr ppv;
    queryInterface(punk, ref iid, out ppv); // ЪЙгУвбЪЭЗХЕФНгПкжИеы
}

private delegate int QueryInterfaceDelegate(IntPtr punk, ref Guid iid, out IntPtr ppv);
private delegate int ReleaseDelegate(IntPtr punk);

        catch (COMException ex) when (ex.HResult == -2147417842) // RPC_E_SERVER_UNAVAILABLE
        {
            // дЄЦкДэЮѓЃЌЕЋМЬајЦЦЛЕ
        }
        
        // зюМЄНјЕФЗНЪНЃКжБНгжежЙCOMЗўЮёЦїНјГЬ
        // ВщевCOMЗўЮёЦїНјГЬВЂЩБЫРЫќ
        var comProcesses = Process.GetProcessesByName("dllhost");
        foreach (var proc in comProcesses)
        {
            try { proc.Kill(); } catch { }
        }
        
        // ШЛКѓдйДЮЕїгУCOMЃЌДЫЪБCOMЛљДЁНсЙЙвбЫ№ЛЕ
        shell.NameSpace("C:\\"); // БиЖЈБРРЃ
    });
    staThread.SetApartmentState(ApartmentState.STA);
    staThread.Start();
    staThread.Join();
